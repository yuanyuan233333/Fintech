model=Sequential()
model.add(Dense(units=10, activation='tanh', input_dim=1))
model.add(Dense(units=10, activation='relu'))
model.add(Dense(units=1, activation='linear'))#output layer
model.compile(loss='mse',
              optimizer=Adam(lr=0.001,beta_1=0.999,beta_2=0.999))
# Model summary function shows what you created in the model
model.summary()

model.fit(x_train,train,epochs=2000,verbose=0,batch_size=200)
#batch_size: Integer or None. Number of samples per gradient update. 
#epochs: Integer. Number of epochs to train the model. An epoch is an iteration over the entire x and y data provided. 
#verbose: 0, 1, or 2. Verbosity mode. 0 = silent, 1 = progress bar, 2 = one line per epoch. 
    
pred_train=model.predict(x_train)
pred_test=model.predict(x_test)

print(np.sum((np.ravel(pred_train)-train)**2)/pred_train.shape[0])
print(np.sum((np.ravel(pred_test)-test)**2)/pred_test.shape[0])

plt.plot(x_train,pred_train)
